class Solution {
    public  boolean buddyStrings(String A, String B) {
        int Alen = A.length();
        int Blen = B.length();
        if(Alen != Blen || Alen <= 1){
            return  false;
        }
        if(A.equals(B)){
            for (int i = 0; i < Alen; i++) {
                if(A.indexOf(A.charAt(i)) != A.lastIndexOf(A.charAt(i))){
                    return true;
                }
            }
        }
        int indexL = 0;
        int indexR = 0;
        boolean flagA = false;
        boolean flagB = false;
            for (int i = 0; i < Alen; i++) {
            if(A.charAt(i) != B.charAt(i)){
                if(flagA == false){
                    indexL = i;
                    flagA = true;
                }else if(flagB == false){
                    indexR = i;
                    flagB = true;
                }else {
                    return  false;
                }
            }
        }
        return flagB == true && A.charAt(indexL) == B.charAt(indexR) && A.charAt(indexR) == B.charAt(indexL);
    }
}
